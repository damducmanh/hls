const simpleArray = "1 2 3 4 5".split(" ").map(num => parseInt(num)); function miniMaxSum(arr) { const [maxValue, minValue] = [Math.max(...arr), Math.min(...arr)] const [maxSum, minSum] = [ arr.reduce((a,b) => a+b) - maxValue, arr.reduce((a,b) => a+b) - minValue ] return `${minSum} ${maxSum}` } function totalArray(arr) { return arr.reduce((partialSum, a) => partialSum + a, 0) } function minInArray(arr) { return Math.min(...arr) } function maxInArray(arr) { return Math.max(...arr); } function findEvenElements(arr) { return arr.filter(num => num % 2 === 0) } function findOddElements(arr) { return arr.filter(num => num % 2 !== 0) } console.log('miniMaxSum',miniMaxSum(simpleArray)); console.log('simpleArray',totalArray(simpleArray)) console.log('minInArray',minInArray(simpleArray)) console.log('maxInArray',maxInArray(simpleArray)) console.log('findEvenElements', findEvenElements(simpleArray)); console.log('findOddElements', findOddElements(simpleArray));
